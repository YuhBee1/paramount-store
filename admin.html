Google Sheets — Orders & Customers
Auto-save every checkout and customer record to your spreadsheet via Google Apps Script.
Not Connected
Quick Setup
  1. Create a Google Sheet with tabs: Customers and Orders
  2. Open Extensions → Apps Script and paste the script below
  3. Click Deploy → New deployment → Web App (Anyone can access)
  4. Copy the Web App URL and paste it here
function doPost(e){
  const d=JSON.parse(e.postData.contents);
  const ss=SpreadsheetApp.getActiveSpreadsheet();
  const sh=ss.getSheetByName(d.sheet)||ss.insertSheet(d.sheet);
  const keys=Object.keys(d.data);
  if(sh.getLastRow()===0) sh.appendRow(['Timestamp',...keys]);
  sh.appendRow([d.timestamp,...keys.map(k=>d.data[k]||'')]);
  return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);
}
function doGet(){return ContentService.createTextOutput('PDS Sheets API Active');}
